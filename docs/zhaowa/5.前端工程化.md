// 高阶函数的范式，在此基础上更改函数的 执行前、执行中、执行后 三个阶段
function HOF0(fn) {
  return function(...args) {
    return fn.apply(this, args);
  }
}

// 函数组合
let compose = (...fns) => (...args) => {
  return fns.reduceRight((param, fn, i) => {
    return i === fns.length-1 ? fn.apply(null, param) : fn.call(null, param)
  }, args)
}
